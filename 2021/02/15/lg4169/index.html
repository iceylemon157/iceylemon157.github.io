[LG4169] 天使的襪子 (CDQ分治)

題目連結

CDQ分治

最近補了一直想要學的CDQ分治。研究了很久才終於稍微搞懂他在幹嘛QQ。CDQ分治主要在處理三維偏序的問題，它主要可以做三件事情：

  1. 解決和點對有關的問題
  2. 優化 $1D/1D$ 動態規劃
  3. 將動態問題轉成靜態問題

我覺得網路上有很多資料(ex:OI wiki)都講得很好了，所以這裡就不放詳細說明了。

昨天在學的時候我總共學了 $1$ 跟 $3$ 兩個用途。$1$ 就是最經典的三維偏序的問題，例如模板 三維偏序 跟 動態逆序對 這兩題。這兩題都只是模板題而已，沒什麼特別的。但是我第二題寫超久QQ，我在寫那題的時候才稍微了解CDQ分治到底可以做什麼。

後來我就去學將動態問題轉成靜態問題的方法。會想要先學這個是因為進階指南裡面有詳解，我覺得講的好像比OI Wiki還要好懂（？。實際上要怎麼做呢？對於區間 $[L, R]$ 而言，會分治到 $[L, mid]$ 跟 $[mid + 1, R]$ 兩個區間。對於橫跨兩邊的答案，要做的是計算左區間的修改對右區間的貢獻就好了。

題意

一開始平面上有 $N$ 個點，總共有 $M$ 次詢問，每一次詢問會有以下兩種操作：

  1. 在平面上新增一個點 $(x,y)$
  2. 詢問離點 $(x,y)$ 最近的點的距離(曼哈頓距離)

題解

先考慮沒有修改的操作。看到題目求的是曼哈頓距離，就代表有絕對值。絕對值是個很麻煩的東西，這題可以考慮分case討論。先只考慮查詢的點在左下角的case，詢問就會等價於求 $\min\limits_{i
\in [1,N],x_i\leq x,y_i\leq y}NaN$，也就是說需要得到 $x_i+y_i$ 的最大值，這就是個二維偏序。對於另外三個case而言，只要把座標變換一下讓他們變到左下角就可以了(透過一些數學)。

到了有修改的時候，每次詢問就變成對於一個詢問 $(x,y)$，找到時間戳在這個詢問之前且 $x_i\leq x,y_i\leq y$ 中 $x_i+y_i$ 的最大值。這樣就變成一個三維偏序的問題了，CDQ分治gogo。用上面提到的方法做就可以了。

心得

我原本想自己解，但解到後來我發現自己還是不太知道要怎麼做修改，我就去看別人的題解。然後我看了一個小時都看不懂別人的程式碼QAQ。我一直不知道他的第二關鍵字怎麼排序的，後來才知道他們都是在做分治的時候順便做merge sort。我好笨QAQQ。這題還有很多比較細節的地方。害我一直錯QQ。

AC Code：

#include<bits/stdc++.h>
#define int long long
#define pb emplace_back
#define INF 0x3f3f3f3f
#define f first
#define s second
#define FOR(i, n) for(int i = 1; i <= n; i ++)
#define FFOR(i, a, b) for(int i = a; i <= b; i ++)
#define pii pair<int, int>
#define mod 1000000007
#define all(x) x.begin(), x.end()
#define mem(x, a) memset(x, a, sizeof(x))
#define loli ios_base::sync_with_stdio(0), cin.tie(0)
using namespace std;
const int maxn = 1e6 + 50;
const int mxN = 1e6 + 5;
int n, m, ans[maxn];
pii p[maxn];

struct node{
    int t, x, y, opt;    
} a[maxn], b[maxn], tem[maxn];

struct BIT {
    int bit[maxn], n;
    void init(int a) {
        n = a;
        FOR(i, n) bit[i] = 0LL;
    }
    int low(int x) { return x & -x; }
    void modi(int i, int v) {
        for(; i <= n; i += low(i)) bit[i] = max(bit[i], v);
    }
    int qry(int i) {
        int ret = 0LL;
        for(; i > 0; i -= low(i)) ret = max(bit[i], ret);
        return ret;
    }
    void clear(int i) {
        for(; i <= n; i += low(i)) bit[i] = 0LL;
    }
} bit;

void cdq(int l, int r) {
    if(l == r) return;
    int mid = (l + r) >> 1;
    cdq(l, mid);
    int L = l;
    int cnt = l;
    FFOR(i, mid + 1, r) {
        while(L <= mid and a[L].x <= a[i].x) {
            if(a[L].opt == 1) bit.modi(a[L].y, a[L].x + a[L].y);
            tem[cnt ++] = a[L];
            L ++;
        }
        if(a[i].opt == 2) {
            int t = bit.qry(a[i].y);
            if(t != 0LL) ans[a[i].t] = min(ans[a[i].t], a[i].x + a[i].y - t);            
        }
        tem[cnt ++] = a[i];
    }
    cdq(mid + 1, r);
    FFOR(i, l, L - 1) if(a[i].opt == 1) bit.clear(a[i].y);
    while(L <= mid) tem[cnt ++] = a[L ++];
    FFOR(i, l, r) a[i] = tem[i];
}

signed main(){
    loli;
    cin >> n >> m;
    FOR(i, n) cin >> b[i].x >> b[i].y, b[i].x ++, b[i].y ++, b[i].t = i, b[i].opt = 1;
    FFOR(i, n + 1, n + m) {
        cin >> b[i].opt >> b[i].x >> b[i].y;
        b[i].x ++, b[i].y ++;
        b[i].t = i;
    }
    n += m;
    auto cmp1 = [](node a, node b) {
        return a.t < b.t;
    };
    mem(ans, INF);
    bit.init(mxN + 1);
    sort(b + 1, b + n + 1, cmp1);
    FOR(i, n) a[i] = b[i];
    cdq(1, n);

    FOR(i, n) a[i] = b[i], a[i].x = mxN - a[i].x;
    cdq(1, n);

    FOR(i, n) a[i] = b[i], a[i].y = mxN - a[i].y;
    cdq(1, n);

    FOR(i, n) a[i] = b[i], a[i].y = mxN - a[i].y, a[i].x = mxN - a[i].x;
    cdq(1, n);

    sort(a + 1, a + n + 1, cmp1);
    FOR(i, n) if(a[i].opt == 2) cout << ans[i] << "\n";
}
